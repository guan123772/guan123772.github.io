数据结构代码汇总

1、线性表

#define MaxNum 100
typedef struct SqList {
Elemtype data[MaxNum];
int length;
}SqList;

2、单链表

typedef struct LNode {
Elemtype data;
struct Node* next;
}LNode,*LinkList;

3、双向链表

typedef struct DNode { Elemtype data; struct DNode* prior,next;
}DNode, * DLinkList;

4、栈

#define MaxNum 100
typedef struct {
Elemtype data[MaxNum];
int top;
}SqStack;

5、队列

#define MaxNum 100
Typedef struct {
ElemType data[MaxNum];
int rear, front;
}SqQueue;

6、串

#define MaxNum 100
typedef struct {
Char ch[MaxNum];
int length;
}SString;

7、二叉树

typedef struct BiTNode{
Elemtype data;
struct BiTNode* lchild, * rchild;
}BitNode,*BitTree;

8、线索二叉树

typedef struct ThreadNode {
ElemType data;
struct BiTNode* lchild, * rchild;
int ltag, rtag;
}ThreadNode,*ThreadTree;

9、树（双亲表示法，孩子兄弟表示法）

（1）双亲表示法

#define MaxNum 100
typedef struct {
Elemtype data;
int Parent;
}PtNode; //节点定义
typedef struct {
PtNode nodes[MaxNum];
int n; //节点数
}PTree; //树的类型定义

（2）孩子兄弟表示法

typedef struct CSnode {
Elemtype data;
struct CSnode*firstchild,*nextsibling //节点数
}CSnode,*CSTree; //树的类型定义

10、图(邻接矩阵法和邻接表法)

（1）邻接矩阵法

#define MaxNum 100
struct char VertexType; //顶点
struct int Edge; //边
typedef struct {
VertexType vex[MaxNum];
EdgeType edge[MaxNum][MaxNum];
int vexnum, arcnum;
}MGraph;

（2）邻接表法

#define MaxNum 100
typedef struct ArcNode{ //边表节点
int adjvex; //顶点位置下标
struct ArcNode* next;
}ArcNode;
typedef struct VNode{ //顶点表节点
VertexType data;
ArcNode* first;
}VNode,Adj[MaxNum];
typedef struct {
AdjList vertices; //邻接表
int vexunm, arcnum;
}AlGraph;
